Documentação importação de dados Infinit Soluções

Guia base para projetos de importação em C#, usando destino Comercial em MySQL, destino Plus em SQL Server, leitura via CSV e origem Firebird quando necessário.

1. Visão geral do padrão

O padrão do projeto é manter uma única classe Importador.cs com métodos separados por entidade, como Cliente(), Produto(), Portador() e GradeProduto().

O Program.cs fica responsável por carregar as conexões, criar os contextos e entregar tudo pronto para o Importador.

2. Criar projeto

No Visual Studio, crie um projeto do tipo Console App.

Tipo de projeto 
Console App
Framework: .NET 8
Nome sugerido: ImportacaoClienteX

O padrão recomendado é .NET 8 com EF Core 8.x para manter compatibilidade.

3. Estrutura padrão do projeto

Use esta estrutura como base:

Estrutura de pastas 
NomeDoProjeto/
│
├── Destino/
│   ├── DestinoContext.cs
│   ├── Produto.cs
│   ├── Pessoa.cs
│   └── demais entidades geradas pelo Scaffold
│
├── Origem/
│   ├── OrigemContext.cs
│   ├── ConnectionFirebird.cs
│   ├── Select.cs
│   └── entidades da origem, quando a origem também for banco
│
├── Importador.cs
├── appsettings.json
└── Program.cs
Quando a origem for CSV, não precisa criar contexto de origem. Nesse caso, o Importador recebe somente o contexto do destino.
Quando a origem for Firebird, coloque os arquivos ConnectionFirebird.cs e Select.cs dentro da pasta Origem.

using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Configuration;
using NomeDoProjeto.Destino;

var configuration = new ConfigurationBuilder()
    .SetBasePath(AppContext.BaseDirectory)
    .AddJsonFile("appsettings.json", optional: false)
    .Build();

string connectionDestino = configuration.GetConnectionString("DestinoComercial");

var optionsDestino = new DbContextOptionsBuilder<DestinoContext>()
    .UseMySql(connectionDestino, ServerVersion.AutoDetect(connectionDestino))
    .Options;

7. Importador.cs

Classe única com construtores para os cenários de CSV, banco origem e Firebird.

Importador.cs - Estrutura base 
using Microsoft.EntityFrameworkCore;
using System.Data;
using NomeDoProjeto.Destino;
using NomeDoProjeto.Origem;

public class Importador
{
    private readonly DestinoContext _destino;
    private readonly OrigemContext? _origem;
    private readonly ConnectionFirebird? _bancoOrigem;

    public Importador(DestinoContext destino)
    {
        _destino = destino;
    }

    public Importador(OrigemContext origem, DestinoContext destino)
    {
        _origem = origem;
        _destino = destino;
    }

    public Importador(DestinoContext destino, ConnectionFirebird bancoOrigem)
    {
        _destino = destino;
        _bancoOrigem = bancoOrigem;
    }

    private static List<string[]> GetCSV(string caminho)
    {
        return File.ReadAllLines(caminho)
            .Skip(1)
            .Where(linha => !string.IsNullOrWhiteSpace(linha))
            .Select(linha => linha.Split(';'))
            .ToList();
    }

    public void GradeProduto()
    {
        List<string[]> grades = GetCSV(@"\\kailon\IMPORTACAO\Importações Victor\Mix More - sistema kigi\backup\produtosGrade.csv");

        var count = grades.Count;
        int contador = 0;

        Console.WriteLine($"Total de Grades encontrados: {count}");

        foreach (var grade in grades)
        {
            contador++;
            Console.WriteLine($"Grade {contador} // {count}");

            var antProdutoId = grade[0].ToString();

            var buscaProduto = _destino.Produtos
                .AsNoTracking()
                .FirstOrDefault(x => x.Alteracaousuario == antProdutoId);

            if (buscaProduto != null)
            {
                var novaGrade = new Grademovimento()
                {
                    AlteracaoData = DateTime.Now,
                    DataCadastro = DateTime.Now,
                    CadastroUsuario = "IMPORTAÇÃO",
                    AlteracaoUsuario = "IMPORTAÇÃO",
                    Tamanho = grade[8],
                    Referencia = grade[4],
                    ProdutoId = buscaProduto.ProdutoId,
                    IsAtivo = true,
                    Cor = grade[7],
                    Quantidade = Convert.ToDecimal(grade[11])
                };

                buscaProduto.Utilizagrade = true;

                _destino.Grademovimentos.Add(novaGrade);
            }

            if (contador % 500 == 0)
            {
                try
                {
                    _destino.SaveChanges();
                    _destino.ChangeTracker.Clear();
                    Console.WriteLine($"{contador} Grades importadas...");
                }
                catch (Exception e)
                {
                    Console.WriteLine(e.Message + "\n" + e.InnerException?.Message);
                }
            }
        }

        _destino.SaveChanges();
        _destino.ChangeTracker.Clear();

        Console.WriteLine("Importação de Grades finalizada.");
    }

    public void GradeProdutoPlusCsv()

    public void ProdutoPlus()
    {
        if (_origem == null)
            throw new InvalidOperationException("OrigemContext não foi informado no construtor.");

        var produtosOrigem = _origem.Produtos
            .AsNoTracking()
            .ToList();

        int total = produtosOrigem.Count;
        int contador = 0;
        int tamanhoBatch = 500;

        Console.WriteLine($"Total de produtos encontrados: {total}");

        foreach (var antProduto in produtosOrigem)
        {
            contador++;

            var novoProduto = new Produto
            {
                Alteracaousuario = antProduto.Alteracaousuario,
                Descricao = antProduto.Descricao,
                Valorvendaatacado = antProduto.Valorvendaatacado,
                Valorvendavarejo = antProduto.Valorvendavarejo,
                Estoqueatual = antProduto.Estoqueatual,
                Referenciaproduto = antProduto.Referenciaproduto,
                Referenciapadrao = antProduto.Referenciapadrao,
                Idativo = antProduto.Idativo,
                Datacadastro = DateTime.Now
            };

            _destino.Produtos.Add(novoProduto);

            if (contador % tamanhoBatch == 0)
            {
                _destino.SaveChanges();
                _destino.ChangeTracker.Clear();
                Console.WriteLine($"Produtos importados: {contador}/{total}");
            }
        }

        _destino.SaveChanges();
        _destino.ChangeTracker.Clear();

        Console.WriteLine($"Importação de produtos finalizada. Total importado: {contador}");
    }

    public void ClienteFirebirdPlus()
    {
        if (_bancoOrigem == null)
            throw new InvalidOperationException("ConnectionFirebird não foi informado no construtor.");

        var clientes = _bancoOrigem.Execute(Select.GetAllFromTable("TCLIENTE"));

        int i = 1;
        int count = clientes.Rows.Count;

        foreach (DataRow row in clientes.Rows)
        {
            Console.WriteLine($"Cliente {i++}/{count}");

            var novoCli = new Pessoa()
            {
                Alteracaodata = DateTime.Now,
                Datacadastro = DateTime.Now,
                Cadastrocompleto = true,
                Cadastrousuario = "IMPORTAÇÃO",
                Alteracaousuario = row["CONTROLE"].ToString(),
                Ativo = row["ATIVO"].ToString() == "SIM" ? true : false,
                Fisjur = row["TIPOCLIENTE"].ToString() == "FÍSICA" ? "F" : "J",
                Iscliente = true,
                Consumidorfinal = 1,
                Nomerazaosocial = row["CLIENTE"].ToString(),
                Nomefantasia = row["FANTASIA"].ToString(),
                Bairro = row["BAIRRO"].ToString(),
                Logradouro = row["ENDERECO"].ToString(),
                Municipio = row["CODIGOCIDADEIBGE"].ToString(),
                Uf = Convert.ToInt32(row["CODIGOCIDADEIBGE"].ToString().Substring(0, 2)),
                Cep = row["CEP"] != DBNull.Value ? FormatarDocumento(row["CEP"].ToString()) : "",
                Cpfcnpj = row["CPF"] != DBNull.Value ? FormatarDocumento(row["CPF"].ToString()) : row["CNPJ"] != DBNull.Value ? FormatarDocumento(row["CNPJ"].ToString()) : "",
                Rgie = row["RG"] != DBNull.Value ? FormatarDocumento(row["RG"].ToString()) : row["IE"] != DBNull.Value ? FormatarDocumento(row["IE"].ToString()) : "",
                Observacao = row["OBS"].ToString(),
                Celular = row["CELULAR"] != DBNull.Value ? FormatarDocumento(row["CELULAR"].ToString()) : "",
            };

            _destino.Pessoas.Add(novoCli);
            _destino.SaveChanges();
        }
    }

    private static string FormatarDocumento(string valor)
    {
        if (string.IsNullOrWhiteSpace(valor))
            return "";

        return new string(valor.Where(char.IsDigit).ToArray());
    }
}

8.2 ConnectionFirebird.cs

ConnectionFirebird.cs - modelo para pasta Origem 
using FirebirdSql.Data.FirebirdClient;
using System.Data;

namespace NomeDoProjeto.Origem
{
    public class ConnectionFirebird
    {
        private string strConnection { get; set; }
        private FbConnection Connection { get; set; }
        private string DbPath { get; set; }
        private FbDataAdapter adapt;
        private FbDataReader reader;

        public ConnectionFirebird()
        {
        }

        public ConnectionFirebird(string dbPath)
        {
            this.DbPath = dbPath;
            this.strConnection =
                "User=SYSDBA;" +
                "Password=masterkey;" +
                "DataBase=localhost:" + DbPath + ";" +
                "Charset=NONE;";

            WorkDoConnection();
        }

        private void WorkDoConnection()
        {
            this.Connection = new FbConnection(strConnection);
            this.reader = null;
            this.adapt = new FbDataAdapter();
        }

        public void OpenConnection()
        {
            this.Connection.Open();
        }

        public void CloseConnection()
        {
            if (this.Connection != null)
            {
                this.Connection.Close();
            }
        }

        public DataTable Execute(string fbSqlQuery)
        {
            DataTable dataTable = new DataTable();

            try
            {
                this.OpenConnection();

                this.adapt = new FbDataAdapter(fbSqlQuery, this.Connection);
                this.adapt.SelectCommand.Transaction = this.Connection.BeginTransaction();
                this.adapt.SelectCommand.ExecuteNonQuery();
                this.adapt.Fill(dataTable);

                return dataTable;
            }
            catch (Exception ex)
            {
                Console.WriteLine(ex.ToString());
                throw;
            }
            finally
            {
                if (Connection.State == ConnectionState.Open)
                    this.CloseConnection();
            }
        }
    }
}

namespace NomeDoProjeto.Origem
{
    public class Select
    {
        public static string GetAllFromTable(string tableName)
        {
            string SQLQuery = string.Format(@"SELECT * FROM {0}", tableName);

            return SQLQuery;
        }

        public static string GetAllFromTableWhere(string tableName, string where)
        {
            string SQLQuery = string.Format(@"SELECT * FROM {0} WHERE {1}", tableName, where);

            return SQLQuery;
        }

        public static string GetTableWithWhere(string tableName, string SQLWhere)
        {
            string SQLQuery = string.Format(@"SELECT * FROM {0} WHERE {1}", tableName, SQLWhere);

            return SQLQuery;
        }

        public static string GetPessoaFeitoCrianca()
        {
            return string.Format(@"
                SELECT 
                    P.cod AS pessoa_cod,
                    P.RAZAOSOCIAL,
                    P.cnpj_cpf,
                    P.insc_Rg,
                    P.status,
                    (SELECT FIRST 1 E.logradouro FROM pes_enderecos E WHERE E.codpessoa = P.cod) AS logradouro,
                    (SELECT FIRST 1 E.bairro FROM pes_enderecos E WHERE E.codpessoa = P.cod) AS bairro,
                    (SELECT FIRST 1 E.cep FROM pes_enderecos E WHERE E.codpessoa = P.cod) AS cep,
                    (SELECT FIRST 1 C.codibge FROM pes_enderecos E LEFT JOIN cep_cidade C ON E.codcepcidade = C.cod WHERE E.codpessoa = P.cod) AS codigoibge,
                    (SELECT FIRST 1 U.sigla FROM pes_enderecos E LEFT JOIN cep_cidade C ON E.codcepcidade = C.cod LEFT JOIN cep_uf U ON C.coduf = U.cod WHERE E.codpessoa = P.cod) AS uf_sigla
                FROM pes_pessoas P;
            ");
        }
    }
}

10. Boas práticas

Use uma única classe Importador.cs com métodos por entidade.
Crie os contextos no Program.cs e passe para o Importador pelo construtor.
Quando a origem for CSV, passe somente o contexto do destino.
Quando a origem for banco, crie Origem e Destino com Scaffold separado.
Use AsNoTracking() em consultas de leitura.
Use SaveChanges em lote, por exemplo a cada 500 registros.
Use ChangeTracker.Clear() após salvar cada lote.
Não edite manualmente as classes geradas pelo Scaffold.
